// 1. 최빈값 계산 함수 (데이터 배열과 정밀도를 인자로 받음) function calculateMode(values, precision = 4) { const frequency = {}; let maxFreq = 0; let modes = []; values.forEach(v => { const val = v.toFixed(precision); frequency[val] = (frequency[val] || 0) + 1; if (frequency[val] > maxFreq) { maxFreq = frequency[val]; } }); for (const val in frequency) { if (frequency[val] === maxFreq) { modes.push(parseFloat(val)); } } // 최빈값이 여러 개일 경우 평균값을 반환하거나 첫 번째 값을 반환 return modes.length > 0 ? modes[0] : null; } // 2. analyzeColumn 함수 수정 function analyzeColumn(colIndex) { const title = rawData[0][colIndex] || `Column ${colIndex + 1}`; const lsl = parseFloat(rawData[1][colIndex]); const usl = parseFloat(rawData[2][colIndex]); const values = rawData.slice(3) .map(row => parseFloat(row[colIndex])) .filter(val => !isNaN(val)); if (values.length < 2) return; // 기본 통계 계산 const n = values.length; const mean = values.reduce((a, b) => a + b, 0) / n; const mode = calculateMode(values); // 최빈값 계산 추가 const sStd = Math.sqrt(values.reduce((a, b) => a + Math.pow(b - mean, 2), 0) / (n - 1)); // 공정능력 지수 const cp = (usl - lsl) / (6 * sStd); const cpk = Math.min((usl - mean) / (3 * sStd), (mean - lsl) / (3 * sStd)); const pStd = Math.sqrt(values.reduce((a, b) => a + Math.pow(b - mean, 2), 0) / n); const pp = (usl - lsl) / (6 * pStd); const ppk = Math.min((usl - mean) / (3 * pStd), (mean - lsl) / (3 * pStd)); addSummaryCard(title, cpk); // UI 생성 (mode 인자 추가) createResultSection(colIndex, title, n, mean, mode, sStd, lsl, usl, cp, cpk, pp, ppk); setTimeout(() => { renderChart(colIndex, mean, mode, sStd, lsl, usl, values); }, 50); } // 3. createResultSection 수정 (Basic Statistics 부분에 Mode 추가) function createResultSection(id, title, n, mean, mode, std, lsl, usl, cp, cpk, pp, ppk) { const section = document.createElement('div'); section.className = 'animate-result'; section.innerHTML = `

${title}

Basic Statistics

${createRow("Data Count (N)", n.toLocaleString())} ${createRow("Mean (μ)", mean.toFixed(4))} ${createRow("Mode (Mo)", mode !== null ? mode.toFixed(4) : "N/A")} ${createRow("Stdev (σ)", std.toFixed(4))} ${createRow("LSL", lsl.toFixed(2))} ${createRow("USL", usl.toFixed(2))}

Process Indices

${createIndexBox("Cp", cp)} ${createIndexBox("Cpk", cpk)} ${createIndexBox("Pp", pp)} ${createIndexBox("Ppk", ppk)}

Normal Distribution Chart

`; resultsWrapper.appendChild(section); } // 4. renderChart 수정 (최빈값 수직선 추가) function renderChart(id, mean, mode, std, lsl, usl, values) { const ctx = document.getElementById(`chart-${id}`).getContext('2d'); // ... (points 생성 로직 동일) new Chart(ctx, { // ... (data 설정 동일) plugins: [{ id: 'limitLines', afterDraw: (chart) => { const { ctx, chartArea: { top, bottom }, scales: { x } } = chart; const drawLine = (val, label, color, isDashed = true, offset = 12) => { const xPos = x.getPixelForValue(val); if (xPos < 0 || xPos > chart.width) return; ctx.save(); ctx.beginPath(); if (isDashed) ctx.setLineDash([5, 5]); ctx.moveTo(xPos, top); ctx.lineTo(xPos, bottom); ctx.lineWidth = 2; ctx.strokeStyle = color; ctx.stroke(); ctx.fillStyle = color; ctx.font = 'bold 10px Inter'; ctx.textAlign = 'center'; ctx.fillText(label, xPos, top - offset); ctx.restore(); }; drawLine(lsl, `LSL: ${lsl}`, '#ff3b30'); drawLine(usl, `USL: ${usl}`, '#ff3b30'); drawLine(mean, `Mean: ${mean.toFixed(2)}`, '#34c759', false); // 최빈값 선 추가 (보라색) if (mode !== null) { drawLine(mode, `Mode: ${mode.toFixed(2)}`, '#af52de', true, 25); } } }] }); }
